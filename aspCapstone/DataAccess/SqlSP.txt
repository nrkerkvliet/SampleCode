CREATE PROCEDURE Usp_InsertUpdateDelete_Customer
@Id int = 0,
@LName varchar(255)=NULL,
@FName varchar(255)=NULL,
@Query int
AS
BEGIN
IF (@Query = 1)
BEGIN
INSERT INTO Accounts (
LName,
FName)
VALUES (
@LName,
@FName)
IF(@@ROWCOUNT > 0)
BEGIN
SELECT 'Insert'
END
END

IF(@Query = 2)
BEGIN
UPDATE Accounts
SET LName = @LName,
FName = @FName
WHERE Accounts.Id = @Id
SELECT 'Update'
END

IF(@Query = 3)
BEGIN
DELETE FROM Accounts WHERE Accounts.Id = @Id
SELECT 'Deleted'
END

IF(@Query = 4)
BEGIN
SELECT * FROM Accounts
END
END

IF(@Query = 5)
BEGIN
SELECT * FROM Accounts
WHERE Accounts.Id = @Id
END
